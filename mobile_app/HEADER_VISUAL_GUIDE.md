### Address Formatting
```kotlin
fun formatAddressForHeader(address: AddressDto): String {
    return buildString {
        if (!address.label.isNullOrBlank()) {
            append(address.label)
            append(" - ")
        }
        append(address.street)
        if (!address.landmark.isNullOrBlank()) {
            append(", ")
            append(address.landmark)
        }
    }
}
```
